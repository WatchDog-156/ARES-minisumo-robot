225void MainWindow::onDataReceived(const QByteArray &data) {
226 QString recivedData = QString::fromUtf8(data).trimmed();
227 qDebug() << "Odebrano z BLE:" << recivedData;
228
229 static const QRegularExpression re("\\d+");
230 QList<int> values;
231 QRegularExpressionMatchIterator i = re.globalMatch(recivedData);
232
233 while (i.hasNext()) {
234 QRegularExpressionMatch match = i.next();
235 values.append(match.captured(0).toInt());
236 }
237
238 if (values.size() == 8) {
239 int lines[2] = {values[0], values[1]};
240 int tofs[4] = {values[2], values[3], values[4], values[5]};
241 int motors[2] = {values[6], values[7]};
242 qDebug() << "Sparsowana wiadomość to:" << lines[0] << ", " << lines[1] << " | " << tofs[0] << ", " << tofs[1] << ", " << tofs[2] << ", " << tofs[3] << " | " << motors[0] << ", " << motors[1];// << std::endl;
243 line->addMeasurement(lines[0],lines[1]);
244 tof->addMeasurement(tofs[0],tofs[1],tofs[2],tofs[3]);
245 } else {
246 qDebug() << "Wiadomość nie została sparsowana";
247 }
248}
