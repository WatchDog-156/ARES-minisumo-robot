80void BluetoothManager::serviceStateChanged(QLowEnergyService::ServiceState s) {
81 if (s == QLowEnergyService::RemoteServiceDiscovered) {
82 writeCharacteristic = service->characteristic(charWriteUuid);
83
84 QLowEnergyCharacteristic readChar = service->characteristic(charReadUuid);
85
86 if (!writeCharacteristic.isValid() || !readChar.isValid()) {
87 emit connectionStatusChanged(false, "Błąd: Nie znaleziono kanałów RX/TX");
88 return;

91 const QLowEnergyDescriptor notificationDesc = readChar.descriptor(QBluetoothUuid::DescriptorType::ClientCharacteristicConfiguration);
92
93 if (notificationDesc.isValid()) {
94 service->writeDescriptor(notificationDesc, QByteArray::fromHex("0100"));
95 }
