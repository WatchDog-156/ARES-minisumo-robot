84void BluetoothScanner::addDevice(const QBluetoothDeviceInfo &info) {
85 if (info.coreConfigurations() & QBluetoothDeviceInfo::LowEnergyCoreConfiguration) {
86 QString label = QString("%1 [%2] (%3 dBm)").arg(info.name(), info.address().toString(), QString::number(info.rssi()));
87
88 if (ui->listDevices->findItems(label, Qt::MatchExactly).isEmpty()) {
89 ui->listDevices->addItem(label);
90 foundDevices.append(info);

102void BluetoothScanner::sortAndRefreshList() {
103 std::sort(foundDevices.begin(), foundDevices.end(), [](const QBluetoothDeviceInfo &a, const QBluetoothDeviceInfo &b) {
104 return a.rssi() > b.rssi();
105 });
106
107 ui->listDevices->clear();
108 for (const auto &device : foundDevices) {
109 QString label = QString("%1 (%2 dBm)").arg(device.name()).arg(device.rssi());
110 ui->listDevices->addItem(label);
111 }
